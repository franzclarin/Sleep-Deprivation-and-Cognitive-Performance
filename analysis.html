Sleep Deprivation and Cognitive Performance

Goals

  • Understand the effects of sleep deprivation (hours of sleep, sleep quality, daytime sleep) and behavioral determinants of health (BMI, caffeine intake, physical activity level) on cognitive performance (attention, working memory, reaction speed, emotion regulation).
  • Model each cognitive performance metric using sleep and behavioral health features.

plot_faceted_binwidth_units <- function(data, var_binwidths, var_units, var_titles,
                                        title, subtitle, fill_color, ncol = 3) {
  vars  <- names(var_binwidths)
  plots <- map(vars, function(v) {
    bw         <- var_binwidths[[v]]
    panel_title <- var_titles[[v]]
    xlab       <- var_units[[v]]
    ggplot(data, aes(x = .data[[v]])) +
      geom_histogram(aes(y = after_stat(density)),
                     binwidth = bw, fill = fill_color,
                     color = "white", alpha = 0.8) +
      geom_density(color = "#C44E52", linewidth = 0.8) +
      labs(title = panel_title, x = xlab, y = "Density") +
      dist_theme +
      theme(
        plot.title   = element_text(face = "bold", size = 10),
        axis.title.x = element_text(size = 8, color = "grey40")
      )
  })

  wrap_plots(plots, ncol = ncol) +
    plot_annotation(
      title    = title,
      subtitle = subtitle,
      theme    = theme(
        plot.title    = element_text(face = "bold", size = 14),
        plot.subtitle = element_text(color = "grey50", size = 11)
      )
    )
}

plot_sleep_vs_cognition <- function(data, sleep_var, sleep_label, sleep_unit) {

  cognitive_vars <- c(
    "Stroop_Task_Reaction_Time",
    "N_Back_Accuracy",
    "Emotion_Regulation_Score",
    "PVT_Reaction_Time"
  )

  # Descriptive panel titles with units
  outcome_labels <- c(
    "Stroop_Task_Reaction_Time" = "Stroop Task Reaction Time (seconds)",
    "N_Back_Accuracy"           = "N-Back Accuracy (% correct)",
    "Emotion_Regulation_Score"  = "Emotion Regulation Score (0–70)",
    "PVT_Reaction_Time"         = "PVT Reaction Time (milliseconds)"
  )

  data |>
    select(all_of(c(sleep_var, cognitive_vars))) |>
    pivot_longer(-all_of(sleep_var), names_to = "outcome", values_to = "value") |>
    mutate(outcome = recode(outcome, !!!outcome_labels)) |>
    ggplot(aes(x = .data[[sleep_var]], y = value)) +
    geom_point(alpha = 0.4, size = 1.8, color = "#4C72B0") +
    geom_smooth(method = "lm", se = TRUE, color = "#C44E52", linewidth = 1) +
    facet_wrap(~ outcome, scales = "free_y", ncol = 2) +
    labs(
      title    = paste("Sleep vs. Cognitive Outcomes:", sleep_label),
      subtitle = "Linear fit with 95% confidence interval",
      x        = sleep_unit,
      y        = NULL
    ) +
    theme_minimal(base_size = 13) +
    theme(
      plot.title       = element_text(face = "bold", size = 15),
      plot.subtitle    = element_text(color = "grey50", size = 11),
      strip.text       = element_text(face = "bold", size = 10),

}
